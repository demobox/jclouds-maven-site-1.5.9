94     @Test
95     public void testDeleteMultipleObjects() {
96        final String request = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n" +
97           "<Delete>" +
98           "<Object><Key>key1</Key></Object>" +
99           "<Object><Key>key2</Key></Object>" +
100          "</Delete>";
101 
102       final Payload requestPayload = Payloads.newStringPayload(request);
103       requestPayload.getContentMetadata().setContentType("text/xml");
104       requestPayload.getContentMetadata().setContentMD5(CryptoStreams.md5(request.getBytes()));
105       
106       final String response = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n" +
107          "<DeleteResult xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\">\n" +
108          "  <Deleted>\n" +
109          "    <Key>key1</Key>\n" +                                          
110          "  </Deleted>\n" +
111          "  <Deleted>\n" +
112          "    <Key>key1.1</Key>\n" +
113          "  </Deleted>\n" +
114          "  <Error>\n" +
115          "    <Key>key2</Key>\n" +
116          "    <Code>AccessDenied</Code>\n" +
117          "    <Message>Access Denied</Message>\n" +
118          "  </Error>\n" +
119          "</DeleteResult>";
120       
121       final Payload responsePayload = Payloads.newStringPayload(response);
122       responsePayload.getContentMetadata().setContentType("text/xml");
123 
124       AWSS3Client client = requestsSendResponses(bucketLocationRequest, bucketLocationResponse,
125          HttpRequest.builder()
126                     .method("POST")
127                     .endpoint("https://test.s3-eu-west-1.amazonaws.com/?delete")
128                     .addHeader("Host", "test.s3-eu-west-1.amazonaws.com")
129                     .addHeader("Date", CONSTANT_DATE)
130                     .addHeader("Authorization", "AWS identity:/k3HQNVVyAQMsr9qhx6hajocVu4=")
131                     .payload(requestPayload)
132                     .build(),
133          HttpResponse.builder()
134                      .statusCode(200)
135                      .addHeader("x-amz-request-id", "7A84C3CD4437A4C0")
136                      .addHeader("Date", CONSTANT_DATE)
137                      .addHeader("ETag", "437b930db84b8079c2dd804a71936b5f")
138                      .addHeader("Server", "AmazonS3")
139                      .payload(responsePayload)
140                      .build()
141       );
142 
143       DeleteResult result = client.deleteObjects("test", ImmutableSet.of("key1", "key2"));
144       assertNotNull(result, "result is null");
145       
146       assertEquals(result.getDeleted(), ImmutableSet.of("key1", "key1.1"));
147       assertEquals(result.getErrors().size(), 1);
148       
149       assertEquals(result.getErrors().get("key2"), new DeleteResult.Error("AccessDenied", "Access Denied"));
150    }
