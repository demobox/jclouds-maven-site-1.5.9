21  import java.util.concurrent.TimeUnit;
22  
23  import org.jclouds.concurrent.Timeout;
24  import org.jclouds.sqs.domain.Action;

26  /**
27   * Provides access to SQS via their REST API.
28   * <p/>
29   * 
30   * @see PermissionAsyncApi
31   * @author Adrian Cole
32   */
33  @Timeout(duration = 30, timeUnit = TimeUnit.SECONDS)
34  public interface PermissionApi {
35  
36     /**
37      * The AddPermission action adds a permission to a queue for a specific
38      * principal. This allows for sharing access to the queue.
39      * 
40      * When you create a queue, you have full control access rights for the
41      * queue. Only you (as owner of the queue) can grant or deny permissions to
42      * the queue. For more information about these permissions, see Shared Queues
43      * in the Amazon SQS Developer Guide.
44      * 
45      * Note
46      * 
47      * AddPermission writes an SQS-generated policy. If you want to write your
48      * own policy, use SetQueueAttributes to upload your policy.
49      * 
50      * @param queue
51      *           queue to change permissions on
52      * @param label
53      * 
54      *           The unique identification of the permission you're setting.
55      *           example: AliceSendMessage
56      * 
57      *           Constraints: Maximum 80 characters; alphanumeric characters,
58      *           hyphens (-), and underscores (_) are allowed.
59      * @param permission
60      *           The action you want to allow for the specified principal.
61      * @param accountId
62      *           The AWS account number of the principal who will be given
63      *           permission. The principal must have an AWS account, but does not
64      *           need to be signed up for Amazon SQS. For information about
65      *           locating the AWS account identification, see Your AWS
66      *           Identifiers in the Amazon SQS Developer Guide.
67      * 
68      *           Constraints: Valid 12-digit AWS account number, without hyphens
69      * 
70      */
71     void addPermissionToAccount(String label, Action permission, String accountId);
72  
73     /**
74      * The RemovePermission action revokes any permissions in the queue policy
75      * that matches the Label parameter. Only the owner of the queue can remove
76      * permissions.
77      * 
78      * @param queue
79      *           queue to change permissions on
80      * 
81      * @param label
82      *           The identification of the permission you want to remove. This is
83      *           the label you added in AddPermission. example: AliceSendMessage
84      */
85     void remove(String label);
86  
87  }
