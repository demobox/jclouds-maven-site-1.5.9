19  package org.jclouds.openstack.nova.v2_0.extensions;

21  import java.util.concurrent.TimeUnit;
22  
23  import org.jclouds.concurrent.Timeout;
24  import org.jclouds.openstack.nova.v2_0.domain.VolumeAttachment;
25  import org.jclouds.openstack.v2_0.ServiceType;
26  import org.jclouds.openstack.v2_0.services.Extension;

28  import com.google.common.annotations.Beta;
29  import com.google.common.collect.FluentIterable;

31  /**
32   * Provides synchronous access to Volume Attachments.
33   * 
34   * This API strictly handles attaching Volumes to Servers. To create and manage Volumes you need to use one of the 
35   * following APIs:
36   * 
37   * 1. The Cinder API
38   *    If your OpenStack deployment is Folsom or later and it supports the Cinder block storage service, use this API.
39   *    @see org.jclouds.openstack.cinder.v1.features.VolumeApi
40   *    
41   * 2. The nova-volume API
42   *    If your OpenStack deployment is Essex or earlier and it supports the nova-volume extension, use this API.
43   *    @see org.jclouds.openstack.nova.v2_0.extensions.VolumeApi
44   * 
45   * @see VolumeAttachmentAsyncApi
46   * @author Everett Toews
47   */
48  @Beta
49  @Extension(of = ServiceType.COMPUTE, namespace = ExtensionNamespaces.VOLUMES)
50  @Timeout(duration = 180, timeUnit = TimeUnit.SECONDS)
51  public interface VolumeAttachmentApi {
52     /**
53      * List Volume Attachments for a given Server.
54      * 
55      * @param serverId The ID of the Server
56      * @return All VolumeAttachments for the Server
57      */
58     FluentIterable<? extends VolumeAttachment> listAttachmentsOnServer(String serverId);
59  
60     /**
61      * Get a specific Volume Attachment for a Volume and Server.
62      * 
63      * @param volumeId The ID of the Volume
64      * @param serverId The ID of the Server
65      * @return The Volume Attachment.
66      */
67     VolumeAttachment getAttachmentForVolumeOnServer(String volumeId, String serverId);
68  
69     /**
70      * Attach a Volume to a Server.
71      * 
72      * Note: If you are using KVM as your hypervisor then the actual device name in the Server will be different than 
73      * the one specified. When the Server sees a new device, it picks the next available name (which in most cases is
74      * /dev/vdc) and the disk shows up there on the Server.
75      * 
76      * @param serverId The ID of the Server
77      * @param volumeId The ID of the Volume
78      * @param device The name of the device this Volume will be identified as in the Server (e.g. /dev/vdc) 
79      * @return The Volume Attachment.
80      */
81     VolumeAttachment attachVolumeToServerAsDevice(String volumeId, String serverId, String device);
82  
83     /**
84      * Detach a Volume from a server.
85      * 
86      * Note: Make sure you've unmounted the volume first. Failure to do so could result in failure or data loss.
87      * 
88      * @param volumeId The ID of the Volume
89      * @param serverId The ID of the Server
90      * @return true if successful
91      */
92     boolean detachVolumeFromServer(String volumeId, String serverId);
93  }
