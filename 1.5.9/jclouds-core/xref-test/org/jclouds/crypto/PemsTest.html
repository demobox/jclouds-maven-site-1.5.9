111    @Test
112    public void testRSAPKCS1RawPublicKeySpecPem() throws IOException, InvalidKeySpecException, NoSuchAlgorithmException {
113       KeySpec spec = Pems.publicKeySpec(Payloads.newStringPayload(PUBLIC_KEY_PKCS1_RAW));
114       String encoded = Pems.pem(KeyFactory.getInstance("RSA").generatePublic(spec));
115       KeySpec generatedSpec = Pems.publicKeySpec(encoded);
116 
117       assertTrue(spec instanceof RSAPublicKeySpec);
118       assertTrue(generatedSpec instanceof RSAPublicKeySpec);
119 
120       // The encoded is different because the generatePublic method adds the
121       // algorithm to the key.
122       assertNotEquals(encoded, PUBLIC_KEY_PKCS1_RAW.replaceAll("\n", "\n").trim());
123       // Verify that the modulus and public exponent of the encoded key are the
124       // same than the ones in the original key
125       assertEquals(RSAPublicKeySpec.class.cast(spec).getModulus(), RSAPublicKeySpec.class.cast(generatedSpec)
126             .getModulus());
127       assertEquals(RSAPublicKeySpec.class.cast(spec).getPublicExponent(), RSAPublicKeySpec.class.cast(generatedSpec)
128             .getPublicExponent());
129    }
