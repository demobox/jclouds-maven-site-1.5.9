386      Optional<URI> endpoint = findEndpoint(method, args);

388      if (!endpoint.isPresent()) {
389         throw new NoSuchElementException(String.format("no endpoint found for %s",
390               new ClassMethodArgs(method.getDeclaringClass(), method, args)));
391      }
392 
393      GeneratedHttpRequest.Builder requestBuilder = GeneratedHttpRequest.builder();
394      HttpRequest r = RestAnnotationProcessor.findHttpRequestInArgs(args);
395      if (r != null) {

399      }
400 
401      // URI template in rfc6570 form
402      UriBuilder uriBuilder = uriBuilder(endpoint.get().toString());
403      
404      overridePathEncoding(uriBuilder, method);
405      
406      requestBuilder.declaring(declaring)
407                    .javaMethod(method)
408                    .args(args)
409                    .caller(caller)
410                    .filters(getFiltersIfAnnotated(method));
411      
412      Multimap<String, Object> tokenValues = LinkedHashMultimap.create();
413 
414      tokenValues.put(Constants.PROPERTY_API_VERSION, apiVersion);
415      tokenValues.put(Constants.PROPERTY_BUILD_VERSION, buildVersion);
416 
417      // make sure any path from the caller is a prefix
418      if (caller != null) {
419         tokenValues.putAll(addPathAndGetTokens(caller.getMethod().getDeclaringClass(), caller.getMethod(),
420               caller.getArgs(), uriBuilder));

424      
425      Multimap<String, Object> formParams;
426      if (caller != null) {
427         formParams = addFormParams(tokenValues, caller.getMethod(), caller.getArgs());
428         formParams.putAll(addFormParams(tokenValues, method, args));
429      } else {
430         formParams = addFormParams(tokenValues, method, args);
431      }
432      Multimap<String, Object> queryParams = addQueryParams(tokenValues, method, args);
433      Multimap<String, String> headers = buildHeaders(tokenValues, method, args);
434 
435      if (r != null)
436         headers.putAll(r.getHeaders());
437 
438      if (shouldAddHostHeader(method)) {
439         StringBuilder hostHeader = new StringBuilder(endpoint.get().getHost());
440         if (endpoint.get().getPort() != -1)
441            hostHeader.append(":").append(endpoint.get().getPort());
442         headers.put(HOST, hostHeader.toString());
443      }
444 
445      Payload payload = null;
446      for(HttpRequestOptions options : findOptionsIn(method, args)) {
447         injector.injectMembers(options);// TODO test case
448         for (Entry<String, String> header : options.buildRequestHeaders().entries()) {
449            headers.put(header.getKey(), replaceTokens(header.getValue(), tokenValues));
450         }
451         for (Entry<String, String> query : options.buildQueryParameters().entries()) {
452            queryParams.put(query.getKey(), replaceTokens(query.getValue(), tokenValues));
453         }
454         for (Entry<String, String> form : options.buildFormParameters().entries()) {
455            formParams.put(form.getKey(), replaceTokens(form.getValue(), tokenValues));
456         }
457 
458         String pathSuffix = options.buildPathSuffix();
459         if (pathSuffix != null) {
460            uriBuilder.appendPath(pathSuffix);
461         }
462         String stringPayload = options.buildStringPayload();
463         if (stringPayload != null)
464            payload = Payloads.newStringPayload(stringPayload);
465      }
466      
467      if (queryParams.size() > 0) {
468         uriBuilder.query(queryParams);
469      }
470 
471      requestBuilder.headers(filterOutContentHeaders(headers));
472 
473      requestBuilder.endpoint(uriBuilder.build(convertUnsafe(tokenValues)));
474      
475      if (payload == null)
476         payload = findPayloadInArgs(args);
477      
478      List<? extends Part> parts = getParts(method, args, ImmutableMultimap.<String, Object> builder()
479                                                                           .putAll(tokenValues)
480                                                                           .putAll(formParams).build());
481      
482      if (parts.size() > 0) {
483         if (formParams.size() > 0) {
484            parts = newLinkedList(concat(transform(formParams.entries(), ENTRY_TO_PART), parts));
485         }
486         payload = new MultipartForm(BOUNDARY, parts);
487      } else if (formParams.size() > 0) {
488         payload = Payloads.newUrlEncodedFormPayload(transformValues(formParams, NullableToStringFunction.INSTANCE));
489      } else if (headers.containsKey(CONTENT_TYPE)) {
490         if (payload == null)
491            payload = Payloads.newByteArrayPayload(new byte[] {});
492         payload.getContentMetadata().setContentType(get(headers.get(CONTENT_TYPE), 0));
493      }
494      if (payload != null) {
495         requestBuilder.payload(payload);
496      }
497      GeneratedHttpRequest request = requestBuilder.build();
498 
499      org.jclouds.rest.MapBinder mapBinder = getMapPayloadBinderOrNull(method, args);
500      if (mapBinder != null) {
501         Map<String, Object> mapParams;
502         if (caller != null) {
503            mapParams = buildPostParams(caller.getMethod(), caller.getArgs());
504            mapParams.putAll(buildPostParams(method, args));
505         } else {
506            mapParams = buildPostParams(method, args);
507         }
508         if (method.isAnnotationPresent(PayloadParams.class)) {
509            PayloadParams params = method.getAnnotation(PayloadParams.class);
510            addMapPayload(mapParams, params, headers);
511         }
512         request = mapBinder.bindToRequest(request, mapParams);
513      } else {
514         request = decorateRequest(request);
515      }
516 
517      if (request.getPayload() != null) {
518         contentMetadataCodec.fromHeaders(request.getPayload().getContentMetadata(), headers);
519      }
520      utils.checkRequestHasRequiredProperties(request);
521      return request;
522   }
