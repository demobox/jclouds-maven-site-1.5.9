34  @Test(groups = "unit", testName = "BindStringAsMetadataValueTest")
35  public class BindStringAsMetadataValueTest {
36     XMLParser xml = new JAXBParser("true");
37  
38     @Test
39     public void testBindMap() {
40        BindStringAsMetadataValue binder = new BindStringAsMetadataValue(xml);
41  
42        HttpRequest request = HttpRequest.builder().method("GET").endpoint("http://momma").build();
43        request = binder.bindToRequest(request, "foo");
44        assertEquals(request.getPayload().getRawContent(), 
45                    XMLParser.DEFAULT_XML_HEADER + "\n"+
46                    "<MetadataValue xmlns=\"http://www.vmware.com/vcloud/v1.5\">" + "\n"+
47                    "    <Value>foo</Value>" + "\n" +
48                    "</MetadataValue>" + "\n");
49        assertEquals(request.getPayload().getContentMetadata().getContentType(), MediaType.APPLICATION_XML);
50     }
