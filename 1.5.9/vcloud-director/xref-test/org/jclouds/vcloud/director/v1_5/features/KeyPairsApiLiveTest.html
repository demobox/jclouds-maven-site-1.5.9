116 	private Media findOrCreateKeyPairContainerInVDCNamed(Vdc currentVDC,
117 			String keyPairsContainerName, final String keyPairName) {
118 		Media keyPairsContainer = null;
119 
120 		Optional<Media> optionalKeyPairsContainer = Iterables.tryFind(
121 				findAllEmptyMediaInOrg(), new Predicate<Media>() {
122 
123 					@Override
124 					public boolean apply(Media input) {
125 						return mediaApi.getMetadataApi(input.getId()).get(
126 								keyPairName) != null;
127 					}
128 				});
129 
130 		if (optionalKeyPairsContainer.isPresent())
131 			keyPairsContainer = optionalKeyPairsContainer.get();
132 
133 		if (keyPairsContainer == null) {
134 			keyPairsContainer = uploadKeyPairInVCD(currentVDC,
135 					keyPairsContainerName, keyPairName);
136 		}
137 		return keyPairsContainer;
138 	}

140 	private Media uploadKeyPairInVCD(Vdc currentVDC,
141 			String keyPairsContainerName, String keyPairName) {
142 		Media keyPairsContainer = addEmptyMediaInVDC(currentVDC,
143 				keyPairsContainerName);
144 		assertNotNull(keyPairsContainer.getFiles(),
145 				String.format(OBJ_FIELD_REQ, MEDIA, "files"));
146 		assertTrue(keyPairsContainer.getFiles().size() == 1, String.format(
147 				OBJ_FIELD_LIST_SIZE_EQ, MEDIA, "files", 1, keyPairsContainer
148 						.getFiles().size()));
149 
150 		Link uploadLink = getUploadLinkForMedia(keyPairsContainer);
151 		// generate an empty iso
152 		byte[] iso = new byte[] {};
153 		context.getApi()
154 				.getUploadApi()
155 				.upload(uploadLink.getHref(), Payloads.newByteArrayPayload(iso));
156 
157 		Checks.checkMediaFor(VCloudDirectorMediaType.MEDIA, keyPairsContainer);
158 		setKeyPairOnkeyPairsContainer(keyPairsContainer, keyPairName, generateKeyPair(keyPairName));
159 
160 		return keyPairsContainer;
161 	}

163 	private Link getUploadLinkForMedia(Media emptyMedia) {
164 		File uploadFile = getFirst(emptyMedia.getFiles(), null);
165 		assertNotNull(uploadFile,
166 				String.format(OBJ_FIELD_REQ, MEDIA, "files.first"));
167 		assertEquals(uploadFile.getSize(), Long.valueOf(0));
168 		assertEquals(uploadFile.getSize().longValue(), emptyMedia.getSize(),
169 				String.format(OBJ_FIELD_EQ, MEDIA, "uploadFile.size()",
170 						emptyMedia.getSize(), uploadFile.getSize()));
171 
172 		Set<Link> links = uploadFile.getLinks();
173 		assertNotNull(links,
174 				String.format(OBJ_FIELD_REQ, MEDIA, "uploadFile.links"));
175 		assertTrue(links.size() >= 1, String.format(OBJ_FIELD_LIST_SIZE_GE,
176 				MEDIA, "uploadfile.links", 1, links.size()));
177 		assertTrue(Iterables.all(links, Predicates.or(
178 				LinkPredicates.relEquals(Link.Rel.UPLOAD_DEFAULT),
179 				LinkPredicates.relEquals(Link.Rel.UPLOAD_ALTERNATE))),
180 				String.format(OBJ_FIELD_REQ, MEDIA, "uploadFile.links.first"));
181 
182 		Link uploadLink = Iterables.find(links,
183 				LinkPredicates.relEquals(Link.Rel.UPLOAD_DEFAULT));
184 		return uploadLink;
185 	}

187 	private Media addEmptyMediaInVDC(Vdc currentVDC, String keyPairName) {
188 		Link addMedia = find(
189 				currentVDC.getLinks(),
190 				and(relEquals("add"), typeEquals(VCloudDirectorMediaType.MEDIA)));
191 
192 		Media sourceMedia = Media.builder().type(VCloudDirectorMediaType.MEDIA)
193 				.name(keyPairName).size(0).imageType(Media.ImageType.ISO)
194 				.description("iso generated as KeyPair bucket").build();
195 
196 		Media emptyMedia = mediaApi.add(addMedia.getHref(), sourceMedia);
197 		Checks.checkMediaFor(MEDIA, emptyMedia);
198 		return emptyMedia;
199 	}

201 	private String generateKeyPair(String keyPairName) {
202 		Map<String, String> key = Maps.newHashMap();
203 		key.put("keyName", keyPairName);
204 		key.put("keyFingerprint", SshKeys.sha1PrivateKey(sshKey.get("private")));
205 		key.put("publicKey", sshKey.get("public"));
206 		return json.toJson(key);
207 	}
