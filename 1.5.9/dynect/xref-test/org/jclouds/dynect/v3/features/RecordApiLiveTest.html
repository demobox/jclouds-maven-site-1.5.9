51     private void checkRecordId(RecordId record) {

58     private void checkRecord(Record<? extends Map<String, Object>> record) {
59        checkRecordId(record);
60        assertTrue(record.getRData().size() > 0, "RData entries should be present for cannot be zero for Record: "
61              + record);
62        checkNotNull(record.getTTL(), "TTL cannot be null for RecordId: %s", record);
63     }

65     @Test
66     protected void testListAndGetRecords() {
67        for (String zone : context.getApi().getZoneApi().list()) {
68           RecordApi api = context.getApi().getRecordApiForZone(zone);
69           ImmutableList<RecordId> records = api.list().toImmutableList();
70           getAnonymousLogger().info("zone: " + zone + " record count: " + records.size());
71  
72           for (RecordId recordId : records) {
73              Record<? extends Map<String, Object>> record;
74              if ("AAAA".equals(recordId.getType())) {
75                 record = checkAAAARecord(api.getAAAA(recordId.getFQDN(), recordId.getId()));
76              } else if ("A".equals(recordId.getType())) {
77                 record = checkARecord(api.getA(recordId.getFQDN(), recordId.getId()));
78              } else if ("CNAME".equals(recordId.getType())) {
79                 record = checkCNAMERecord(api.getCNAME(recordId.getFQDN(), recordId.getId()));
80              } else if ("MX".equals(recordId.getType())) {
81                 record = checkMXRecord(api.getMX(recordId.getFQDN(), recordId.getId()));
82              } else if ("NS".equals(recordId.getType())) {
83                 record = checkNSRecord(api.getNS(recordId.getFQDN(), recordId.getId()));
84              } else if ("PTR".equals(recordId.getType())) {
85                 record = checkPTRRecord(api.getPTR(recordId.getFQDN(), recordId.getId()));
86              } else if ("SOA".equals(recordId.getType())) {
87                 record = checkSOARecord(api.getSOA(recordId.getFQDN(), recordId.getId()));
88              } else if ("SRV".equals(recordId.getType())) {
89                 record = checkSRVRecord(api.getSRV(recordId.getFQDN(), recordId.getId()));
90              } else if ("TXT".equals(recordId.getType())) {
91                 record = checkTXTRecord(api.getTXT(recordId.getFQDN(), recordId.getId()));
92              } else {
93                 record = api.get(recordId);
94              }
95              assertEquals(record, recordId);
96              checkRecord(record);
97           }
98        }
99     }
