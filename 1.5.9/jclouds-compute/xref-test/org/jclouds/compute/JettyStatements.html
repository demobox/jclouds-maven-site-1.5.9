19  package org.jclouds.compute;
20  
21  import static org.jclouds.scriptbuilder.domain.Statements.exec;
22  import static org.jclouds.scriptbuilder.domain.Statements.extractTargzAndFlattenIntoDirectory;
23  
24  import java.net.URI;
25  
26  import org.jclouds.scriptbuilder.domain.Statement;
27  import org.jclouds.scriptbuilder.domain.StatementList;
28  import org.jclouds.scriptbuilder.statements.java.InstallJDK;
29  import org.jclouds.scriptbuilder.statements.login.AdminAccess;
30  
31  /**
32   * 
33   * @author Adrian Cole
34   */
35  public class JettyStatements {
36  
37     public static final URI JETTY_URL = URI.create(System.getProperty("test.jetty-url",//
38           "http://download.eclipse.org/jetty/8.1.5.v20120716/dist/jetty-distribution-8.1.5.v20120716.tar.gz"));
39  
40     public static final String JETTY_HOME = "/usr/local/jetty";
41     
42     public static final int port = 8080;
43  
44     public static Statement version() {
45        return exec(String.format("head -1 %s/VERSION.txt | cut -f1 -d ' '", JETTY_HOME));
46     }
47  
48     public static Statement install() {
49        return new StatementList(
50              AdminAccess.builder().adminUsername("web").build(),
51              InstallJDK.fromOpenJDK(),
52              authorizePortInIpTables(),
53              extractTargzAndFlattenIntoDirectory(JETTY_URL, JETTY_HOME),
54              exec("chown -R web " + JETTY_HOME));
55     }
56  
57     private static Statement authorizePortInIpTables() {
58        return new StatementList(
59                 exec("iptables -I INPUT 1 -p tcp --dport " + port + " -j ACCEPT"),
60                 exec("iptables-save"));
61     }
62     
63     public static Statement start() {
64        return new StatementList(
65              exec("cd " + JETTY_HOME),
66              exec("./bin/jetty.sh start"));
67     }
68     
69     public static Statement stop() {
70        return new StatementList(
71              exec("cd " + JETTY_HOME),
72              exec("./bin/jetty.sh stop"));
73     }
74  }
