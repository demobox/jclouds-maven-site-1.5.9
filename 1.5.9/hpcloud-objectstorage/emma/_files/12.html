132   private HttpRequest signForTemporaryAccess(HttpRequest request, long timeInSeconds) {
133      Credentials currentCreds = checkNotNull(creds.get(), "credential supplier returned null");
134      // accessKey is of the form tenantName:accessKeyId (not tenantId)
135      String accessKeyId = currentCreds.identity.substring(currentCreds.identity.indexOf(':') + 1);
136      String secretKey = currentCreds.credential;
137      String tenantId = access.get().getToken().getTenant().get().getId();
138      
139      HttpRequest.Builder builder = request.toBuilder();
140      // HP Cloud does not use X-Auth-Token for temporary signed URLs and
141      // leaking this allows clients arbitrary privileges until token timeout.
142      builder.filters(filter(request.getFilters(), not(instanceOf(AuthenticateRequest.class))));
