21  import static com.google.common.base.Preconditions.checkNotNull;
22  
23  import java.net.URI;
24  import java.net.URISyntaxException;
25  import java.util.Map;

27  import javax.ws.rs.core.MediaType;

29  import org.jclouds.fujitsu.fgcp.reference.RequestParameters;
30  import org.jclouds.http.HttpRequest;
31  import org.jclouds.rest.MapBinder;
32  import org.jclouds.rest.binders.BindToStringPayload;
33  
34  import com.google.common.base.Strings;

36  /**
37   * Not currently used but leaving for reference when implementing multipart POST
38   * methods.
39   * 
40   * @author Dies Koper
41   */
42  public class BindParamsToXmlPayload extends BindToStringPayload implements
43        MapBinder {
44  
45     @SuppressWarnings("unchecked")
46     @Override
47     public <R extends HttpRequest> R bindToRequest(R request,
48           Map<String, Object> mapParams) {
49        String action = checkNotNull(
50              mapParams.remove(RequestParameters.ACTION),
51              RequestParameters.ACTION).toString();
52        String version = Strings.nullToEmpty((String) mapParams
53              .remove(RequestParameters.VERSION));
54  
55        StringBuilder xml = new StringBuilder();
56        xml.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\r\n");
57        xml.append("<OViSSRequest>\r\n");
58        xml.append("  <Action>" + action + "</Action>\r\n");
59  
60        for (Map.Entry<String, Object> entry : mapParams.entrySet()) {
61           String key = entry.getKey();
62           xml.append("  <" + key + ">" + checkNotNull(mapParams.get(key))
63                 + "</" + key + ">\r\n");
64        }
65  
66        xml.append("  <Version>" + version + "</Version>\r\n");
67        xml.append("  <Locale></Locale>\r\n"); // value inserted in
68                                      // RequestAuthenticator#filter
69        xml.append("  <AccessKeyId></AccessKeyId>\r\n"); // value inserted in
70                                             // RequestAuthenticator#filter
71        xml.append("  <Signature></Signature>\r\n"); // value inserted in
72                                          // RequestAuthenticator#filter
73        xml.append("</OViSSRequest>");
74  
75        request = super.bindToRequest(request, xml);
76        request.getPayload().getContentMetadata()
77              .setContentType(MediaType.TEXT_XML);
78  
79        // remove version query param if set as it was moved to the xml body
80        URI uri = request.getEndpoint();
81        URI uriWithoutQueryParams;
82        try {
83           uriWithoutQueryParams = new URI(uri.getScheme(), uri.getUserInfo(),
84                 uri.getHost(), uri.getPort(), uri.getPath(), null,
85                 uri.getFragment());
86        } catch (URISyntaxException e) {
87           // should never happen as we're copying the components from a URI
88           uriWithoutQueryParams = uri;
89        }
90  
91        return (R) request.toBuilder().endpoint(uriWithoutQueryParams).build();
92     }
93  
94     @Override
95     public <R extends HttpRequest> R bindToRequest(R request, Object toBind) {
96        throw new IllegalArgumentException(
97              "BindParamsToXmlPayload needs bind parameters");
98     }
99  }
