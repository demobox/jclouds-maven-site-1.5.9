30  /**
31   * Provides access to Cloud Files via their REST API.

36  @Timeout(duration = 120, timeUnit = TimeUnit.SECONDS)
37  public interface CloudFilesClient extends SwiftClient {

39     /**
40      * Retrieve a list of existing CDN-enabled containers.
41      */
42     Set<ContainerCDNMetadata> listCDNContainers(ListCdnContainerOptions... options);

44     /**
45      * Get the CDN attributes of the container.
46      * 
47      * If the container is (or ever has been) CDN-enabled, the URL, TTL, enabled status, and log retention status are 
48      * returned in the response headers. Its CDN URL can be combined with any object name within the container to form
49      * the publicly accessible URL for that object for distribution over a CDN system. The TTL value is the number of
50      * seconds that the object will be cached in the CDN system before being refetched. The enabled status indicates
51      * whether the container is currently marked to allow public serving of objects via CDN. The log_retention setting
52      * specifies whether the CDN access logs should be collected and stored in the Cloud Files storage system.
53      */
54     ContainerCDNMetadata getCDNMetadata(String container);

56     /**
57      * Before a container can be CDN-enabled, it must exist in the storage system. When a container is CDN-enabled, any
58      * objects stored in it are publicly accessible over the Content Delivery Network by combining the container's CDN
59      * URL with the object name.
60      * 
61      * Any CDN-accessed objects are cached in the CDN for the specified amount of time called the TTL, or Time to Live.
62      * The default TTL value is 259200 seconds, or 72 hours. Each time the object is accessed after the TTL expires, the
63      * CDN refetches and caches the object for the TTL period.
64      * 
65      * You specify the TTL for an object via the ttl parameter. Setting the TTL is the same as setting the HTTP Expires
66      * and Cache-Control headers for the cached object. The minimum TTL is 15 minutes and the maximum is 1 year for a
67      * range of 900 to 31536000 seconds. Setting a TTL for a long time, however, does not guarantee that the content
68      * will stay populated on CDN edge servers for the entire period. The most popular objects stay cached based on the
69      * edge location's logic.    
70      */
71     URI enableCDN(String container, long ttl, boolean logRetention);

74      * @see CloudFilesClient#enableCDN(String, long, boolean)

76     URI enableCDN(String container, long ttl);

98     /**
99      * Remove the container from the CDN. Please note, however, that objects remain public until their TTL expires.
100     */
101    boolean disableCDN(String container);

103    /**
104     * You can purge a CDN-enabled object when you find it absolutely necessary to remove the object from public access
105     * and you cannot wait for the TTL to expire. You should limit object purges to situations where there could be
106     * serious personal, business, or security consequences if it remained in the CDN. For example, someone published
107     * your company's quarterly earnings too early. You can manually purge CDN-enabled objects without having to wait
108     * for the TTL to expire, and you can optionally be notified by email that the object has been purged. However, you
109     * may only DELETE up to 25 objects per day. Any attempt to delete more than this will result in a 498 status code
110     * error (Rate Limited).
111     * 
112     * There are two ways you may purge objects from the edge: (1) individually using purgeCDNObject() in the API or
113     * (2) by creating a support ticket to purge entire containers. The 25-object limit does not apply when purging an
114     * entire container via Support.    
115     */
116    boolean purgeCDNObject(String container, String object, Iterable<String> emails);
117    
118    /**
119     * @see CloudFilesClient#purgeCDNObject(String, String, Iterable)
120     */
121    boolean purgeCDNObject(String container, String object);

123    /**
124     * You may use your Cloud Files account to create a static website on the World Wide Web. First, you must CDN-enable
125     * a storage container. Any HTML or static web pages in the container will become available through a static website
126     * once you set the parameter index to "index.html" or other index page of your choice. You may also create
127     * subdirectories in your website by creating pseudo-directories (objects with a '/' in the name).
128     * 
129     * The page you set for index becomes the index page for every subdirectory in your website; each of your
130     * pseudo-directories should contain a file with that name. So, if you set index to "index.html", you should have an
131     * index.html page in each pseudo-directory. If you do not have the named index page, visits to myhost/subdir/ will
132     * return a 404 error.
133     * 
134     * To setup a domain name for your static website, create a CNAME with your DNS Server (or name server). This is the
135     * domain name of your site (such as www.example.com). Your CNAME is set up with your individual DNS Server, which
136     * is outside the scope of this documentation. Once you have your CNAME established, map your domain name to your
137     * Cloud Files CDN URL to get your site up and running on the Web.    
138     */
139    boolean setCDNStaticWebsiteIndex(String container, String index);
140 
141    /**
142     * You may create and set custom error pages for visitors to your website; currently, only 401 (Unauthorized) and
143     * 404 (Not Found) errors are supported. To do this, set the error parameter.
144     * 
145     * Error pages are served with the <status> code prepended to the name of the error page you set. For instance, if
146     * you set error to "error.html", 401 errors will display the page 401error.html. Similarly, 404 errors will display
147     * 404error.html. You must have both of these pages created in your container when you set the error parameter, or
148     * your site will display generic error pages.
149     * 
150     * You need only set the error parameter once for your entire static website.    
151     */
152    boolean setCDNStaticWebsiteError(String container, String error);
