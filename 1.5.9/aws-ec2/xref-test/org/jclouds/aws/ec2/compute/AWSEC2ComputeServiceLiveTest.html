92     @Override
93     @Test
94     public void testExtendedOptionsAndLogin() throws Exception {
95        String region = "us-west-2";
96  
97        RestContext<IAMApi, IAMAsyncApi> iamContext = ContextBuilder.newBuilder(new AWSIAMProviderMetadata())
98                                                                    .credentials(identity, credential)
99                                                                    .modules(setupModules()).build();
100       String profileName = "ec2Test";
101       // Note this needs to wait an undefined amount of time before it is visible to ec2, seems to be about 15seconds
102       InstanceProfile profile = createIAMInstanceProfile(iamContext.getApi(), profileName);
103 
104       AWSSecurityGroupClient securityGroupClient = AWSEC2Client.class.cast(
105                view.unwrap(AWSEC2ApiMetadata.CONTEXT_TOKEN).getApi()).getSecurityGroupServices();
106 
107       KeyPairClient keyPairClient = EC2Client.class.cast(view.unwrap(AWSEC2ApiMetadata.CONTEXT_TOKEN).getApi())
108                .getKeyPairServices();
109 
110       InstanceClient instanceClient = EC2Client.class.cast(view.unwrap(AWSEC2ApiMetadata.CONTEXT_TOKEN).getApi())
111                .getInstanceServices();
112 
113       String group = this.group + "o";
114 
115       Date before = new Date();
116 
117       ImmutableMap<String, String> userMetadata = ImmutableMap.<String, String> of("Name", group);
118       ImmutableSet<String> tags = ImmutableSet. of(group);
119 
120       // note that if you change the location, you must also specify image parameters
121       Template template = client.templateBuilder().locationId(region).osFamily(AMZN_LINUX).os64Bit(true).build();
122       template.getOptions().userMetadata(userMetadata);
123       template.getOptions().as(AWSEC2TemplateOptions.class).enableMonitoring();
124       template.getOptions().as(AWSEC2TemplateOptions.class).spotPrice(0.3f);
125       template.getOptions().as(AWSEC2TemplateOptions.class).iamInstanceProfileName(profileName);
126 
127       String startedId = null;
128       try {
129          cleanupExtendedStuffInRegion(region, securityGroupClient, keyPairClient, group);
130 
131          Thread.sleep(3000);// eventual consistency if deletes actually occurred.
132 
133          // create a security group that allows ssh in so that our scripts later
134          // will work
135          String groupId = securityGroupClient.createSecurityGroupInRegionAndReturnId(region, group, group);
136 
137          securityGroupClient.authorizeSecurityGroupIngressInRegion(region, groupId, permit(IpProtocol.TCP).port(22));
138 
139          template.getOptions().as(AWSEC2TemplateOptions.class).securityGroupIds(groupId);
140 
141          // create a keypair to pass in as well
142          KeyPair result = keyPairClient.createKeyPairInRegion(region, group);
143          template.getOptions().as(AWSEC2TemplateOptions.class).keyPair(result.getKeyName());
144 
145          // pass in the private key, so that we can run a script with it
146          assert result.getKeyMaterial() != null : result;
147          template.getOptions().overrideLoginPrivateKey(result.getKeyMaterial());
148 
149          Set<? extends NodeMetadata> nodes = client.createNodesInGroup(group, 1, template);
150          NodeMetadata first = getOnlyElement(nodes);
151 
152          // Name metadata should turn into node.name
153          assertEquals(first.getName(), group);
154 
155          checkUserMetadataInNodeEquals(first, userMetadata);
156          checkTagsInNodeEquals(first, tags);
157 
158          assert first.getCredentials() != null : first;
159          assert first.getCredentials().identity != null : first;
160 
161          startedId = first.getProviderId();
162 
163          AWSRunningInstance instance = AWSRunningInstance.class.cast(getOnlyElement(getOnlyElement(instanceClient
164                   .describeInstancesInRegion(region, startedId))));
165 
166          assertEquals(instance.getKeyName(), group);
167          assert instance.getSpotInstanceRequestId() != null;
168          assertEquals(instance.getMonitoringState(), MonitoringState.ENABLED);
169          assertEquals(instance.getIAMInstanceProfile().get().getArn(), profile.getArn());
170 
171          // generate some load
172          ListenableFuture<ExecResponse> future = client.submitScriptOnNode(first.getId(), Statements
173                   .exec("while true; do true; done"), runAsRoot(false).nameTask("cpuSpinner"));
174 
175          // monitoring granularity for free tier is 5 minutes, so lets make sure we have data.
176          Thread.sleep(TimeUnit.MILLISECONDS.convert(5, TimeUnit.MINUTES));
177 
178          // stop the spinner
179          future.cancel(true);
180 
181          RestContext<CloudWatchApi, CloudWatchAsyncApi> monitoringContext = ContextBuilder
182                .newBuilder(new AWSCloudWatchProviderMetadata())
183                .credentials(identity, credential)
184                .modules(setupModules()).build();
185 
186          try {
187             GetMetricStatisticsResponse datapoints = monitoringContext.getApi().getMetricApiForRegion(instance.getRegion())
188                      .getMetricStatistics(GetMetricStatistics.builder()
189                                                              .dimension(new Dimension(EC2Constants.Dimension.INSTANCE_ID, instance.getId()))
190                                                              .unit(Unit.PERCENT)
191                                                              .namespace("AWS/EC2")
192                                                              .metricName("CPUUtilization")
193                                                              .startTime(before)
194                                                              .endTime(new Date())
195                                                              .period(60)
196                                                              .statistic(Statistics.AVERAGE)
197                                                              .build());
198             assert (datapoints.size() > 0) : instance;
199          } finally {
200             monitoringContext.close();
201          }
202 
203          // make sure we made our dummy group and also let in the user's group
204          assertEquals(newTreeSet(instance.getGroupNames()), ImmutableSortedSet.<String> of("jclouds#" + group, group));
205 
206          // make sure our dummy group has no rules
207          SecurityGroup secgroup = getOnlyElement(securityGroupClient.describeSecurityGroupsInRegion(instance
208                   .getRegion(), "jclouds#" + group));
209 
210          assert secgroup.getIpPermissions().size() == 0 : secgroup;
211 
212          // try to run a script with the original keyPair
213          runScriptWithCreds(group, first.getOperatingSystem(), LoginCredentials.builder().user(
214                   first.getCredentials().identity).privateKey(result.getKeyMaterial()).build());
215 
216       } finally {
217          client.destroyNodesMatching(NodePredicates.inGroup(group));
218          if (startedId != null) {
219             // ensure we didn't delete these resources!
220             assertEquals(keyPairClient.describeKeyPairsInRegion(region, group).size(), 1);
221             assertEquals(securityGroupClient.describeSecurityGroupsInRegion(region, group).size(), 1);
222          }
223          deleteIAMInstanceProfile(iamContext.getApi(), profileName);
224          iamContext.close();
225          cleanupExtendedStuffInRegion(region, securityGroupClient, keyPairClient, group);
226       }

229    static String assumeRolePolicy = "{\"Version\":\"2008-10-17\",\"Statement\":[{\"Sid\":\"\",\"Effect\":\"Allow\",\"Principal\":{\"Service\":\"ec2.amazonaws.com\"},\"Action\":\"sts:AssumeRole\"}]}";
230    static String route53Policy = "{\"Statement\":[{\"Effect\":\"Allow\",\"Action\":\"route53:*\",\"Resource\":\"*\"}]}";
231 
232    static InstanceProfile createIAMInstanceProfile(IAMApi api, String name) {
233       String roleName = name + "route53";
234       api.getRoleApi().createWithPolicy(roleName, assumeRolePolicy);
235       api.getPolicyApiForRole(roleName).create("route53-readonly", route53Policy);
236       api.getInstanceProfileApi().create(name);
237       api.getInstanceProfileApi().addRole(name, roleName);
238       InstanceProfile updated = api.getInstanceProfileApi().get(name);
239       getAnonymousLogger().info("created instanceProfile: " + updated);
240       return updated;
241    }
242 
243    static void deleteIAMInstanceProfile(IAMApi api, String name) {
244       String roleName = name + "route53";
245       api.getInstanceProfileApi().removeRole(name, roleName);
246       api.getPolicyApiForRole(roleName).delete("route53-readonly");
247       api.getRoleApi().delete(roleName);
248       api.getInstanceProfileApi().delete(name);
249    }
